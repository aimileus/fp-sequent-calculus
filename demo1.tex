\begin{prooftree}
\hypo{p_{1},p_{2}\Rightarrow p_{2}}
\hypo{p_{1},p_{2}\Rightarrow p_{1}}
\infer2[\(\wedge R\)]{p_{1},p_{2}\Rightarrow p_{1}\wedge p_{2}}

\infer1[\(\Box L\)]{\Box (p_{2}),p_{1}\Rightarrow p_{1}\wedge p_{2}}

\infer1[\(\Box L\)]{\Box (p_{1}),\Box (p_{2})\Rightarrow p_{1}\wedge p_{2}}

\infer1[\(\Box R\)]{\Box (p_{1}),\Box (p_{2})\Rightarrow \Box (p_{1}\wedge p_{2})}

\infer1[\(\Diamond R\)]{\Box (p_{1}),\Box (p_{2})\Rightarrow \Diamond (\Box (p_{1}\wedge p_{2}))}

\infer1[\(\Diamond L\)]{\Box (p_{1}),\Diamond (\Box (p_{2}))\Rightarrow \Diamond (\Box (p_{1}\wedge p_{2}))}

\infer1[\(\Box L\)]{\Box (\Diamond (\Box (p_{2}))),\Box (p_{1})\Rightarrow \Diamond (\Box (p_{1}\wedge p_{2}))}

\infer1[\(\Diamond L\)]{\Box (\Diamond (\Box (p_{2}))),\Diamond (\Box (p_{1}))\Rightarrow \Diamond (\Box (p_{1}\wedge p_{2}))}

\infer1[\(\Box L\)]{\Box (\Diamond (\Box (p_{1}))),\Box (\Diamond (\Box (p_{2})))\Rightarrow \Diamond (\Box (p_{1}\wedge p_{2}))}

\infer1[\(\Box R\)]{\Box (\Diamond (\Box (p_{1}))),\Box (\Diamond (\Box (p_{2})))\Rightarrow \Box (\Diamond (\Box (p_{1}\wedge p_{2})))}

\end{prooftree}
